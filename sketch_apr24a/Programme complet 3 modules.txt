#include <WiFi.h>
#include <WebServer.h>

// =======================
// WIFI
// =======================
const char* ssid = "SFR_1F50";
const char* password = "pus28urzq7fwj94uxpp9";

WebServer server(80);

// =======================
// 74HC595 (SORTIES)
// =======================
#define DATA_PIN 12
#define CLOCK_PIN 23
#define LATCH_PIN 22

// =======================
// 74HC165 (ENTREES)
// =======================
#define IN_LOAD  0
#define IN_CLOCK 2
#define IN_DATA  15

// =======================
// CAPTEURS
// =======================
#define CAPTEUR_PORTAIL   5
#define CAPTEUR_PORTILLON 18
#define CAPTEUR_GARAGE    19

byte lastInputs = 0;

// =======================
// SECURITE BOOT
// =======================
unsigned long bootTime = 0;
const unsigned long BOOT_DELAY = 5000;

// =======================
// ETAT SYSTEME
// =======================
bool busy = false;
int currentRelay = -1;

int cycle = 0;
int bitIndex = 0;

unsigned long lastStepTime = 0;
unsigned long pulseStartTime = 0;

int pulseDuration = 0;
bool pulsePhase = false;

// =======================
// 74HC595
// =======================
void writeByte(byte data) {
  digitalWrite(LATCH_PIN, LOW);

  for (int i = 7; i >= 0; i--) {
    digitalWrite(CLOCK_PIN, LOW);
    digitalWrite(DATA_PIN, (data >> i) & 0x01);
    digitalWrite(CLOCK_PIN, HIGH);
  }

  digitalWrite(LATCH_PIN, HIGH);
}

void clearAll() {
  writeByte(0x00);
  delayMicroseconds(50);
  writeByte(0x00);
}

// =======================
// 74HC165
// =======================
byte readInputs165() {

  byte value = 0;

  digitalWrite(IN_LOAD, LOW);
  delayMicroseconds(5);
  digitalWrite(IN_LOAD, HIGH);

  for (int i = 0; i < 8; i++) {
    value <<= 1;
    value |= digitalRead(IN_DATA);

    digitalWrite(IN_CLOCK, HIGH);
    delayMicroseconds(5);
    digitalWrite(IN_CLOCK, LOW);
  }

  return ~value;
}

// =======================
// CAPTEURS
// =======================
int readCapteur(int pin) {
  return !digitalRead(pin); // inversion logique (INPUT_PULLUP)
}

// =======================
// RELAIS
// =======================
int getPulseDuration(int relay) {
  if (relay == 2) return 5000;
  if (relay == 1 || relay == 3) return 500;
  return 300;
}

void startRelay(int relay) {

  // 🔒 sécurité post-boot
  if (millis() - bootTime < BOOT_DELAY) {
    Serial.println("SECURITE BOOT ACTIVE");
    return;
  }

  if (busy) return;

  Serial.print("Start relais ");
  Serial.println(relay);

  busy = true;
  currentRelay = relay;

  cycle = 0;
  bitIndex = 0;

  pulseDuration = getPulseDuration(relay);
  pulsePhase = false;

  lastStepTime = millis();
}

// =======================
// MACHINE RELAIS
// =======================
void updateRelay() {

  if (!busy) {
    writeByte(0x00);
    return;
  }

  unsigned long now = millis();

  if (pulsePhase) {
    if (now - pulseStartTime >= pulseDuration) {
      pulsePhase = false;
      cycle++;
      bitIndex = 0;
    }
    return;
  }

  if (now - lastStepTime < 2) return;
  lastStepTime = now;

  if (bitIndex == 7) {
    writeByte(0x00);
  } else {
    writeByte(1 << bitIndex);
  }

  if (bitIndex == currentRelay && cycle == 4) {
    pulsePhase = true;
    pulseStartTime = now;
    return;
  }

  bitIndex++;

  if (bitIndex >= 8) {
    bitIndex = 0;
    cycle++;

    if (cycle >= 5) {
      clearAll();
      busy = false;
      currentRelay = -1;

      Serial.println("Relais terminé");
    }
  }
}

// =======================
// ACTIONS
// =======================
void triggerPortail() { startRelay(1); }
void triggerPortillon() { startRelay(2); }
void triggerGarage() { startRelay(3); }

// =======================
// INPUTS 74HC165
// =======================
void handleInputs() {

  byte state = readInputs165();

  byte changed = state ^ lastInputs;
  byte pressed = changed & state;

  if (pressed & (1 << 0)) triggerPortail();
  if (pressed & (1 << 1)) triggerPortillon();
  if (pressed & (1 << 2)) triggerGarage();

  if (pressed & (1 << 3)) triggerPortail();
  if (pressed & (1 << 4)) triggerPortillon();
  if (pressed & (1 << 5)) triggerGarage();

  lastInputs = state;
}

// =======================
// CORS
// =======================
void sendCORS() {
  server.sendHeader("Access-Control-Allow-Origin", "*");
  server.sendHeader("Access-Control-Allow-Methods", "GET, POST, OPTIONS");
  server.sendHeader("Access-Control-Allow-Headers", "*");
}

// =======================
// ROUTES HTTP
// =======================
void handlePortail() {
  triggerPortail();
  sendCORS();
  server.send(200, "text/plain", "OK");
}

void handlePortillon() {
  triggerPortillon();
  sendCORS();
  server.send(200, "text/plain", "OK");
}

void handleGarage() {
  triggerGarage();
  sendCORS();
  server.send(200, "text/plain", "OK");
}

void handleStatus() {
  sendCORS();

  int portail = readCapteur(CAPTEUR_PORTAIL);
  int portillon = readCapteur(CAPTEUR_PORTILLON);
  int garage = readCapteur(CAPTEUR_GARAGE);

  String status = String(portail) + "," +
                  String(portillon) + "," +
                  String(garage);

  server.send(200, "text/plain", status);
}

void handleOptions() {
  sendCORS();
  server.send(204);
}

// =======================
// SETUP
// =======================
void setup() {
  Serial.begin(115200);

  // 74HC595
  pinMode(DATA_PIN, OUTPUT);
  pinMode(CLOCK_PIN, OUTPUT);
  pinMode(LATCH_PIN, OUTPUT);

  digitalWrite(DATA_PIN, LOW);
  digitalWrite(CLOCK_PIN, LOW);
  digitalWrite(LATCH_PIN, LOW);

  clearAll();

  // 74HC165
  pinMode(IN_LOAD, OUTPUT);
  pinMode(IN_CLOCK, OUTPUT);
  pinMode(IN_DATA, INPUT);

  digitalWrite(IN_LOAD, HIGH);
  digitalWrite(IN_CLOCK, LOW);

  // CAPTEURS
  pinMode(CAPTEUR_PORTAIL, INPUT_PULLUP);
  pinMode(CAPTEUR_PORTILLON, INPUT_PULLUP);
  pinMode(CAPTEUR_GARAGE, INPUT_PULLUP);

  // timer boot
  bootTime = millis();

  // WIFI
  WiFi.begin(ssid, password);

  while (WiFi.status() != WL_CONNECTED) {
    delay(300);
    Serial.print(".");
  }

  Serial.println("\nWiFi OK");
  Serial.println(WiFi.localIP());

  // ROUTES
  server.on("/portail", handlePortail);
  server.on("/portillon", handlePortillon);
  server.on("/garage", handleGarage);
  server.on("/status", handleStatus);

  server.on("/portail", HTTP_OPTIONS, handleOptions);
  server.on("/portillon", HTTP_OPTIONS, handleOptions);
  server.on("/garage", HTTP_OPTIONS, handleOptions);
  server.on("/status", HTTP_OPTIONS, handleOptions);

  server.begin();
}

// =======================
// LOOP
// =======================
void loop() {

  server.handleClient();

  // 🔒 sécurité post-boot
  if (millis() - bootTime < BOOT_DELAY) {
    writeByte(0x00);
    return;
  }

  handleInputs();
  updateRelay();
}
