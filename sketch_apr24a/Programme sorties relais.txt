#include <WiFi.h>
#include <WebServer.h>

// =======================
// WIFI
// =======================
const char* ssid = "SFR_1F50";
const char* password = "pus28urzq7fwj94uxpp9";

WebServer server(80);

// =======================
// 74HC595
// =======================
#define DATA_PIN 12
#define CLOCK_PIN 23
#define LATCH_PIN 22

// =======================
// ETAT SYSTEME
// =======================
bool busy = false;
int currentRelay = -1;

int cycle = 0;
int bitIndex = 0;

unsigned long lastStepTime = 0;
unsigned long pulseStartTime = 0;

int pulseDuration = 0;
bool pulsePhase = false;

// =======================
// ETAT (status API)
// =======================
int portail = 0;
int portillon = 0;
int garage = 0;

// =======================
// SHIFT REGISTER
// =======================
void writeByte(byte data) {
  digitalWrite(LATCH_PIN, LOW);

  for (int i = 7; i >= 0; i--) {
    digitalWrite(CLOCK_PIN, LOW);
    digitalWrite(DATA_PIN, (data >> i) & 0x01);
    digitalWrite(CLOCK_PIN, HIGH);
  }

  digitalWrite(LATCH_PIN, HIGH);
}

// =======================
// CLEAR TOTAL
// =======================
void clearAll() {
  writeByte(0x00); // 🔥 garantit aucun relais actif
}

// =======================
// DUREE RELAIS
// =======================
int getPulseDuration(int relay) {
  if (relay == 2) return 5000;        // portillon
  if (relay == 1 || relay == 3) return 500; // portail + garage
  return 300;
}

// =======================
// DEMARRAGE RELAIS
// =======================
void startRelay(int relay) {
  if (busy) return;

  Serial.print("Start relais ");
  Serial.println(relay);

  busy = true;
  currentRelay = relay;

  cycle = 0;
  bitIndex = 0;

  pulseDuration = getPulseDuration(relay);
  pulsePhase = false;

  lastStepTime = millis();
}

// =======================
// MACHINE NON BLOQUANTE
// =======================
void updateRelay() {
  if (!busy) return;

  unsigned long now = millis();

  // Phase impulsion longue
  if (pulsePhase) {
    if (now - pulseStartTime >= pulseDuration) {
      pulsePhase = false;
      cycle++;
      bitIndex = 0;
    }
    return;
  }

  // Cadence rapide (~2ms)
  if (now - lastStepTime < 2) return;
  lastStepTime = now;

  byte data = (1 << bitIndex);
  writeByte(data);

  // Déclenche impulsion
  if (bitIndex == currentRelay && cycle == 4) {
    pulsePhase = true;
    pulseStartTime = now;
    return;
  }

  bitIndex++;

  if (bitIndex >= 8) {
    bitIndex = 0;
    cycle++;

    if (cycle >= 5) {
      clearAll();          // 🔥 coupe tout proprement
      writeByte(0x00);     // 🔥 sécurité anti latch

      busy = false;
      currentRelay = -1;

      Serial.println("Relais terminé");
    }
  }
}

// =======================
// ACTIONS
// =======================
void triggerPortail() {
  startRelay(1);
  portail = !portail;
}

void triggerPortillon() {
  startRelay(2);
  portillon = !portillon;
}

void triggerGarage() {
  startRelay(3);
  garage = !garage;
}

// =======================
// CORS
// =======================
void sendCORS() {
  server.sendHeader("Access-Control-Allow-Origin", "*");
  server.sendHeader("Access-Control-Allow-Methods", "GET, POST, OPTIONS");
  server.sendHeader("Access-Control-Allow-Headers", "*");
}

// =======================
// ROUTES HTTP
// =======================
void handlePortail() {
  triggerPortail();
  sendCORS();
  server.send(200, "text/plain", "OK");
}

void handlePortillon() {
  triggerPortillon();
  sendCORS();
  server.send(200, "text/plain", "OK");
}

void handleGarage() {
  triggerGarage();
  sendCORS();
  server.send(200, "text/plain", "OK");
}

void handleStatus() {
  sendCORS();

  String status = String(portail) + "," +
                  String(portillon) + "," +
                  String(garage);

  server.send(200, "text/plain", status);
}

void handleOptions() {
  sendCORS();
  server.send(204);
}

// =======================
// SETUP
// =======================
void setup() {
  Serial.begin(115200);
  Serial.println("ESP32 DEMARRAGE");

  // 74HC595
  pinMode(DATA_PIN, OUTPUT);
  pinMode(CLOCK_PIN, OUTPUT);
  pinMode(LATCH_PIN, OUTPUT);
  clearAll();

  // WIFI
  WiFi.begin(ssid, password);

  Serial.print("Connexion WiFi");
  while (WiFi.status() != WL_CONNECTED) {
    delay(300);
    Serial.print(".");
  }

  Serial.println("\nWiFi OK");
  Serial.println(WiFi.localIP());

  // ROUTES
  server.on("/portail", HTTP_GET, handlePortail);
  server.on("/portillon", HTTP_GET, handlePortillon);
  server.on("/garage", HTTP_GET, handleGarage);
  server.on("/status", HTTP_GET, handleStatus);

  // CORS
  server.on("/portail", HTTP_OPTIONS, handleOptions);
  server.on("/portillon", HTTP_OPTIONS, handleOptions);
  server.on("/garage", HTTP_OPTIONS, handleOptions);
  server.on("/status", HTTP_OPTIONS, handleOptions);

  server.begin();
  Serial.println("Serveur HTTP démarré");
}

// =======================
// LOOP
// =======================
void loop() {
  server.handleClient();
  updateRelay(); // 🔥 coeur non bloquant
}
